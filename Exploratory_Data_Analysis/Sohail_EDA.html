# removing revenue from company size and mnaking it na
dsPosting_2025 <-dsPosting_2025 %>%
  mutate(company_size = ifelse(str_detect(company_size, "€"), NA, company_size))


# removing public from revenue and mnaking it na
dsPosting_2025 <-dsPosting_2025 %>%
  mutate( revenue = ifelse(str_detect(revenue, "Public"), NA, company_size))

# removing private from revenue and mnaking it na
dsPosting_2025 <-dsPosting_2025 %>%
  mutate( revenue = ifelse(str_detect(revenue, "Private"), NA, company_size))

dsPosting_2025 <- dsPosting_2025 %>%
  mutate(salary = str_remove_all(salary, "€"))
 

# converting the euro to dollar in revenue and salary and naming salaryusd
dsPosting_2025 <- dsPosting_2025 %>%
  mutate(
    salary_min   = as.numeric(str_extract(dsPosting_2025, "^[0-9]+")),
    salary_max   = as.numeric(str_extract(dsPosting_2025, "(?<=– )[0-9]+")),
    # Use average for ranges, otherwise just the single value
    salary_num   = if_else(is.na(salary_max), salary_min, (salary_min + salary_max) / 2),
    avg_salary_usd   = salary_num * 1.1254
  ) 

dsPosting_2025 <-dsPosting_2025 |> 
  select(job_title, seniority_level,status, location, post_date, industry, ownership, company_size,revenue,salary_clean, avg_salary_usd, skills )
